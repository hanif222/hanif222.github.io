TEXTBOX

Membuat sebuah textbox cukup mudah. Kita tinggal membuat sebuah object. Lalu menambahkan beberapa event berikut.

CREATE EVENT

textbox0 = hanif_textbox_create(32, 32, 32 * 18, -1); //-1 untuk default
hanif_textbox_set_label(textbox0, "Masukkan nama anda");

nama = "";

STEP EVENT

hanif_textbox_step(textbox0);

nama = hanif_textbox_get_value(textbox0);

DRAW EVENT

draw_text(0, 0, "Nama anda adalah: " + nama);

DRAW GUI EVENT

hanif_textbox_draw(textbox0);

PASSWORD

Menjadikan textbox ke bentuk password cukup mudah. Kita tinggal memanggil fungsi berikut.

hanif_textbox_password(textbox0, true);

TOMBOL

Membuat sebuah tombol menjadi sangat mudah. Kita tinggal membuat sebuah object. Lalu menambahkan beberapa event berikut.

CREATE EVENT

button0 = hanif_button_create(64, 48, 32 * 4.5, 32, "Tombol 0");
button1 = hanif_button_create(64, 112, 32 * 4.5, 32 * 3.5, "Tombol 1");
button2 = hanif_button_create(64, 272, 32 * 16.5, 32, "Tombol 2");

STEP EVENT

if(hanif_button_step(button0)){
show_message("Tombol 0 ditekan!");
}
else if(hanif_button_step(button1)){
show_message("Tombol 1 ditekan!");
}
else if(hanif_button_step(button2)){
show_message("Tombol 2 ditekan!");
}

DRAW GUI EVENT

hanif_button_draw(button0);
hanif_button_draw(button1);
hanif_button_draw(button2);
